from typing import List, Dict

def detect_muling_cycles(transactions_graph: nx.DiGraph, depth_limit: int = 5) -> List[Dict]:
    """
    Scans the directed transaction graph for closed loops (cycles)
    which indicate circular fund routing often used in money laundering.
    """
    suspicious_rings = []
    
    # Find all simple cycles in the directed graph
    cycles = list(nx.simple_cycles(transactions_graph))
    
    for ring in cycles:
        # Filter out standard 2-node refunds, look for multi-layer loops
        if 2 < len(ring) <= depth_limit:
            
            risk_score = calculate_ring_velocity(transactions_graph, ring)
            
            if risk_score > 0.85:
                suspicious_rings.append({
                    'nodes': ring,
                    'pattern': 'Circular Routing',
                    'risk_severity': 'CRITICAL',
                    'velocity_score': risk_score
                })
                
    return suspicious_rings
